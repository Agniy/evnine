361 	/** isHasCrossBrowserMinimumVersion()<br />
362 		* 
363 		*  Функция проверки. <br />
364 		*  Является ли версия браузера минимальной для работы к с якорной навигацией.
365 		* 
366 		* @access public
367 		* @return {boolean}
368 		*/
369 	function isHasCrossBrowserMinimumVersion() {
370 		if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,1)+$EVNINE_NAME+"isHasCrossBrowserMinimumVersion() BEGIN");
371 		var userAgent = navigator.userAgent.toLowerCase();
372 		jQuery.browser = {
373 			'version': (userAgent.match( /.+(?:rv|it|ra|ie|me)[\/: ]([\d.]+)/ ) || [])[1],
374 			'chrome': (/chrome/).test( userAgent ),
375 			'safari': (/webkit/).test( userAgent ) && !(/chrome/).test( userAgent ),
376 			'opera': (/opera/).test( userAgent ),
377 			'msie': (/msie/).test( userAgent ) && !(/opera/).test( userAgent ),
378 			'mozilla': (/mozilla/).test( userAgent ) && !(/(compatible|webkit)/).test( userAgent )
379 		};
380 		/**
381 		 *  Поддерживает ли браузер работу с evNav плагином?
382 		 * 
383 		 * @var boolean
384 		 * @access public
385 		 */
386 		var $return=true;
387 		jQuery.each($options.crossBrowserMinimumVersion, function($browser,$version){
388 			if (jQuery.browser[$browser]){
389 				if (getInt(jQuery.browser.version)>=$version){
390 					if ($options.debugToConsole) console.warn("#getInt(jQuery.browser.version): "+getInt(jQuery.browser.version));
391 					if ($options.debugToConsole) console.warn("#$version: "+$version);
392 					if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,2)+$EVNINE_NAME+"isHasCrossBrowserMinimumVersion return true");
393 					$return=true;
394 					return '';
395 				}else {
396 					if ($options.debugToConsole) console.warn("#getInt(jQuery.browser.version): "+getInt(jQuery.browser.version));
397 					if ($options.debugToConsole) console.warn("#$version: "+$version);
398 					if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,2)+$EVNINE_NAME+"isHasCrossBrowserMinimumVersion return false");
399 					$return=false;
400 					return '';
401 				}
402 			}
403 		});
404 		if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,1)+$EVNINE_NAME+"isHasCrossBrowserMinimumVersion() END");
405 		return $return;
406 	}

429 	/** getClickHref($that)<br />
430 		* 
431 		*  Выполнить AJAX запрос исходя из типа кликнутого элемента
432 		* 
433 		* @param {jQuery object} $that
434 		* @access public
435 		* @return {boolean}
436 		*/
437 	function getClickHref($that){
438 		if (jQuery($that).attr('href')!=undefined){
439 			$href = jQuery($that).attr('href');
440 			if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,3)+$EVNINE_NAME+"getClickHref $href: "+$href);
441 			$reg = new RegExp($options.isHREFMatchThisRegExpSetNoUseAJAX,"g");
442 			if ($href.match($reg)){
443 			/**
444 				*  Случай когда не нужно использовать AJAX
445 				*/
446 				if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,4)+$EVNINE_NAME+"getClickHref is no ajax href");
447 				if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,4)+$EVNINE_NAME+"getClickHref match $options.isHREFMatchThisRegExpSetNoUseAJAX");
448 				return true;
449 			}else {
450 			/**
451 				*  Используем для обработки AJAX GET запроса при клике на ссылку
452 				*/
453 				getAJAXHref($href);
454 				if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,4)+$EVNINE_NAME+"getClickHref is ajax href");
455 			}
456 		}else if (jQuery($that).attr('type')==='submit'){
457 		/**
458 			*  Если кликнули на отправку формы через AJAX POST
459 			*/
460 			if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,4)+$EVNINE_NAME+"getClickHref is ajax submit");
461 				getAJAXSubmit($that);
462 		}
463 		return false;
