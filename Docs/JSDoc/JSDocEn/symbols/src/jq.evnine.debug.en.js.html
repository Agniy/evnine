 12 	/**
 13 		*  Return the indentation<br />
 14 		*  Example Call<br />
 15 		* this.getTab('| ',0)=''<br />
 16 		* this.getTab('| ',1)='| '<br />
 17 		* this.getTab('| ',2)='| | '
 18 		*
 19 		* @param {string} [$debugPrefixString='']
 20 		*  Debug prefix for group of functions<br />
 21 		*
 22 		* @param {int} [$shift_int='']
 23 		*  Base shift to show nesting<br />
 24 		* 
 25 		* @return {string} $tab=''
 26 		* <br /> return the indentation
 27 		*/
 28 	getTab:function($debugPrefixString,$shift_int){
 29 		if ($shift_int===0){
 30 			return '';
 31 		}
 32 		var $tab='';
 33 		$i=1;
 34 		while($i <= $shift_int) {
 35 			$i++;
 36 			$tab=$tab+$debugPrefixString;
 37 		}
 38 		return $tab;
 39 	},
 40 
 41 	/** 
 42 	 *  Get stack trace<br />
 43 	 *  (c) Eric Wendelin, eriwen.com/javascript<br />
 44 	 * 
 45 	 * @return object
 46 	 */
 47 	getTraceFunction:function() {
 48 		var callstack = [];
 49 		var isCallstackPopulated = false;
 50 		try {
 51 			/**
 52 			*  doesn't exist- that's the point
 53 			*/
 54 			i.dont.exist+=0; 
 55 		} catch(e) {
 56 			var lines = e.stack.split('\n');
 57 			if (e.stack) { 
 58 			/**
 59 				*  Firefox
 60 				*/
 61 				for (i=0, len=lines.length; i<len; i++) {
 62 					if (lines[i].match(/^\s*[A-Za-z0-9\-_\$]+\(/)) {
 63 						lines[i] = lines[i].replace(/@.*|(?=showResponse).*/,"");
 64 						callstack.push(lines[i]);
 65 					}
 66 				}
 67 				/**
 68 					*  Remove call to printStackTrace()
 69 					*/
 70 				callstack.shift();
 71 				isCallstackPopulated = true;
 72 			}
 73 			else if (window.opera && e.message) { //Opera
 74 				for (i=0, len=lines.length; i<len; i++) {
 75 					if (lines[i].match(/^\s*[A-Za-z0-9\-_\$]+\(/)) {
 76 						var entry = lines[i];
 77 						/**
 78 							*  Append next line also since it has the file info
 79 							*/
 80 						if (lines[i+1]) {
 81 							entry += ' at ' + lines[i+1];
 82 							i++;
 83 						}
 84 						callstack.push(entry);
 85 					}
 86 				}
 87 				/**
 88 					*  Remove call to printStackTrace()
 89 					*/
 90 				callstack.shift();
 91 				isCallstackPopulated = true;
 92 			}
 93 		}
 94 			/**
 95 				*  IE and Safari
 96 				*/
 97 			if (!isCallstackPopulated) {
 98 			var currentFunction = arguments.callee.caller;
 99 			while (currentFunction) {
100 				var fn = currentFunction.toString();
101 				var fname = fn.substring(fn.indexOf("function") + 8, fn.indexOf('')) || 'anonymous';
102 				callstack.push(fname);
103 				currentFunction = currentFunction.caller;
104 			}
105 		}
106 		return callstack.reverse();
107 	},
108 
109 		/** 
110 		*  List contents of an object with spaces for debugging<br />
111 		* 
112 		* @param {object} [$obj=undefined]
113 		*  Object to output<br />
114 		*
115 		* @param {string} [$tab='']
116 		* getTraceObject($obj[prop],$tab=$tab+$shift);
117 		*  Indent, which changes the derivation of children objects<br />
118 		* 
119 		* @param {string} [$shift=' ']
120 		*  Base shift to show nesting<br />
121 		* 
122 		* @param {boolean} [$show_function=false]
123 		*  Derive methods in the object
124 		* 
125 		* @return void
126 		*/
127 	getTraceObject:function($obj,$tab,$shift,$show_function){
128 		if ($show_function===undefined){
129 			$show_function=false;
130 		}
131 		var s = "";
132 		if ($shift==undefined){
133 			$shift=' ';
134 		}
135 		if ($tab===undefined) $tab='';
136 		for (prop in $obj)
137 		{
138 			$typeof = typeof $obj[prop];
139 			if ($typeof != "function" && $typeof != "object")
140 			{
141 				s = $tab+"[" + prop + "] => " + $obj[prop] + "";
142 				console.info(s);
143 			}else if($typeof === "object"){
144 					s = $tab+"[" + prop + "] => "+typeof $obj[prop]+" (";
145 					console.info(s);
146 					this.getTraceObject($obj[prop],$tab+$shift);
147 					s= $tab+")";
148 					console.info(s);
149 			}else if($typeof === "function"){
150 				if ($show_function){
151 					console.warn($tab+"[" + prop + "] => "+$obj[prop].toString().split("\n").join("\n"+$tab));
152 				}else {
153 					console.info($tab+"[" + prop + "] => function ");
154 				}
155 			}
156 		}
157 	},
158 
159 	/**
160 		*  Function for the join functions group in the console<br />
161 		* 
162 		* @param {boolean} [$debugToConsoleNotSupport=false]
163 		*  Use a buffer for the console and call alert()<br />
164 		* 
165 		* @return void
166 		*/
167 	initGroupFunctionCall:function($debugToConsoleNotSupport){
168 		if (!$debugToConsoleNotSupport){
169 			/**
170 			 *  Replace the basic techniques of working with the console<br />
171 			 */
172 			window.console.warn_old=window.console.warn;
173 			window.console.info_old=window.console.info;
174 			window.console.warn=function($str){
175 				if ($str.match(/END/)){
176 					console.groupEnd();
177 				}else if ($str.match(/BEGIN$/)){
178 					console.group($str);
179 				}else {
180 					window.console.warn_old($str);
181 				}
182 			};
183 			window.console.info=function($str){
184 				window.console.info_old($str);
185 			};
186 		}
187 	},
188 		
189 	/**
190 		*  If clients browser is not supported by the console, use an 
191 		* <br /> alternative way of collecting all .warn and .info in the buffer<br />
192 		* 
193 		* @param {boolean} [$debugToConsoleNotSupport=false]
194 		*  Use a buffer for the console and call alert()<br />
195 		* @return void
196 		*/
197 	initNotSupport:function($debugToConsoleNotSupport){
198 		if ($debugToConsoleNotSupport){
199 			/**
200 			 *  Create a new object to work with the console<br />
201 			 */
202 			window.console=new Object();
203 			var debug_buffer='';
204 			window.console.warn=function($str){
205 				alert('INFO:'+debug_buffer+"\n\r"+'WARN:'+"\n\r\t"+$str);
206 				debug_buffer='';
207 			};
208 			window.console.info=function($str){
209 				debug_buffer+="\n\r"+$str;
210 			};
211 			window.console.error=function($str){
212 				debug_buffer+="\n\rERROR:"+$str;
213 			};
214 		}
215 	}
