 67 	$show_function=false;

 96 /**

 99 getTraceFunction:function() {
100 	var callstack = [];
101 	var isCallstackPopulated = false;
102 	try {
103 		i.dont.exist+=0; //doesn't exist- that's the point
104 	} catch(e) {
105 		var lines = e.stack.split('\n');
106 		if (e.stack) { //Firefox
107 			for (i=0, len=lines.length; i<len; i++) {
108 				if (lines[i].match(/^\s*[A-Za-z0-9\-_\$]+\(/)) {
109 					lines[i] = lines[i].replace(/@.*|(?=showResponse).*/,"");
110 					callstack.push(lines[i]);
111 				}
112 			}
113 			//Remove call to printStackTrace()
114 			callstack.shift();
115 			isCallstackPopulated = true;
116 		}
117 		else if (window.opera && e.message) { //Opera
118 			for (i=0, len=lines.length; i<len; i++) {
119 				if (lines[i].match(/^\s*[A-Za-z0-9\-_\$]+\(/)) {
120 					var entry = lines[i];
121 					//Append next line also since it has the file info
122 					if (lines[i+1]) {
123 						entry += ' at ' + lines[i+1];
124 						i++;
125 					}
126 					callstack.push(entry);
127 				}
128 			}
129 			//Remove call to printStackTrace()
130 			callstack.shift();
131 			isCallstackPopulated = true;
132 		}
133 	}
134 	if (!isCallstackPopulated) { //IE and Safari
135 		var currentFunction = arguments.callee.caller;
136 		while (currentFunction) {
137 			var fn = currentFunction.toString();
138 			var fname = fn.substring(fn.indexOf("function") + 8, fn.indexOf('')) || 'anonymous';
139 			callstack.push(fname);
140 			currentFunction = currentFunction.caller;
141 		}
142 	}
143 	return callstack.reverse();
144 }
