1 //<script type="text/javascript">
  2 new function (document, $, undefined) {
  3 /** 
  4 	* @class jQuery.evDev
  5 	* <br /> Плагин для отладки кода.
  6 	* <br />
  7 	* <br />
  8 	* <br /> Двойная лицензия MIT или GPL v.2 
  9 	*/
 10 jQuery.evDev={
 11 	/**
 12 		*  Создаём отступы<br />
 13 		*  Пример вызова<br />
 14 		* this.getTab('| ',0)=''<br />
 15 		* this.getTab('| ',1)='| '<br />
 16 		* this.getTab('| ',2)='| | '
 17 		*
 18 		* @param {string} [$debugPrefixString='']
 19 		*  Префикс для вывода в окно отладки группирования по функциям (FireFox FireBug, Chrome, Opera)
 20 		*
 21 		* @param {int} [$shift_int='']
 22 		*  Начальный сдвиг для отображения вложенности
 23 		* 
 24 		* @return {string} $tab=''
 25 		* <br /> вернем отступы
 26 		*/
 27 	getTab:function($debugPrefixString,$shift_int){
 28 		if ($shift_int===0){
 29 			return '';
 30 		}
 31 		var $tab='';
 32 		$i=1;
 33 		while($i <= $shift_int) {
 34 			$i++;
 35 			$tab=$tab+$debugPrefixString;
 36 		}
 37 		return $tab;
 38 	},
 39 
 40 	/** 
 41 	 *  Отследить вызов функции<br />
 42 	 *  (c) Эрик Венделин, eriwen.com/javascript<br />
 43 	 * 
 44 	 * @return object
 45 	 */
 46 	getTraceFunction:function() {
 47 		var callstack = [];
 48 		var isCallstackPopulated = false;
 49 		try {
 50 			/**
 51 			*/
 52 			i.dont.exist+=0; 
 53 		} catch(e) {
 54 			var lines = e.stack.split('\n');
 55 			if (e.stack) { 
 56 			/**
 57 				*/
 58 				for (i=0, len=lines.length; i<len; i++) {
 59 					if (lines[i].match(/^\s*[A-Za-z0-9\-_\$]+\(/)) {
 60 						lines[i] = lines[i].replace(/@.*|(?=showResponse).*/,"");
 61 						callstack.push(lines[i]);
 62 					}
 63 				}
 64 				/**
 65 					*/
 66 				callstack.shift();
 67 				isCallstackPopulated = true;
 68 			}
 69 			else if (window.opera && e.message) { //Opera
 70 				for (i=0, len=lines.length; i<len; i++) {
 71 					if (lines[i].match(/^\s*[A-Za-z0-9\-_\$]+\(/)) {
 72 						var entry = lines[i];
 73 						/**
 74 							*/
 75 						if (lines[i+1]) {
 76 							entry += ' at ' + lines[i+1];
 77 							i++;
 78 						}
 79 						callstack.push(entry);
 80 					}
 81 				}
 82 				/**
 83 					*/
 84 				callstack.shift();
 85 				isCallstackPopulated = true;
 86 			}
 87 		}
 88 			/**
 89 				*/
 90 			if (!isCallstackPopulated) {
 91 			var currentFunction = arguments.callee.caller;
 92 			while (currentFunction) {
 93 				var fn = currentFunction.toString();
 94 				var fname = fn.substring(fn.indexOf("function") + 8, fn.indexOf('')) || 'anonymous';
 95 				callstack.push(fname);
 96 				currentFunction = currentFunction.caller;
 97 			}
 98 		}
 99 		return callstack.reverse();
100 	},
101 
102 		/** 
103 		*  Вывести содержимое объекта с отступами для отладки
104 		* 
105 		* @param {object} [$obj=undefined]
106 		*  Объект для вывода
107 		*
108 		* @param {string} [$tab='']
109 		* getTraceObject($obj[prop],$tab=$tab+$shift);
110 		*  Отступ, который изменяется при выводе вложенных объектов
111 		* 
112 		* @param {string} [$shift=' ']
113 		*  Начальный сдвиг для отображения вложенности
114 		* 
115 		* @param {boolean} [$show_function=false]
116 		*  Вывести методы в объекте
117 		* 
118 		* @return void
119 		*/
120 	getTraceObject:function($obj,$tab,$shift,$show_function){
121 		if ($show_function===undefined){
122 			$show_function=false;
123 		}
124 		var s = "";
125 		if ($shift==undefined){
126 			$shift=' ';
127 		}
128 		if ($tab===undefined) $tab='';
129 		for (prop in $obj)
130 		{
131 			$typeof = typeof $obj[prop];
132 			if ($typeof != "function" && $typeof != "object")
133 			{
134 				s = $tab+"[" + prop + "] => " + $obj[prop] + "";
135 				console.info(s);
136 			}else if($typeof === "object"){
137 					s = $tab+"[" + prop + "] => "+typeof $obj[prop]+" (";
138 					console.info(s);
139 					this.getTraceObject($obj[prop],$tab+$shift);
140 					s= $tab+")";
141 					console.info(s);
142 			}else if($typeof === "function"){
143 				if ($show_function){
144 					console.warn($tab+"[" + prop + "] => "+$obj[prop].toString().split("\n").join("\n"+$tab));
145 				}else {
146 					console.info($tab+"[" + prop + "] => function ");
147 				}
148 			}
149 		}
150 	},
151 
152 	/**
153 		*  Функция вывода группировки в консоле
154 		* 
155 		* @param {boolean} [$debugToConsoleNotSupport=false]
156 		*  Использовать буфер для консоли и делать alert()
157 		* 
158 		* @return void
159 		*/
160 	initGroupFunctionCall:function($debugToConsoleNotSupport){
161 		if (!$debugToConsoleNotSupport){
162 			/**
163 			 *  Заменяем базовые методы работы с консолью
164 			 */
165 			window.console.warn_old=window.console.warn;
166 			window.console.info_old=window.console.info;
167 			window.console.warn=function($str){
168 				if ($str.match(/END/)){
169 					console.groupEnd();
170 				}else if ($str.match(/BEGIN$/)){
171 					console.group($str);
172 				}else {
173 					window.console.warn_old($str);
174 				}
175 			};
176 			window.console.info=function($str){
177 				window.console.info_old($str);
178 			};
179 		}
180 	},
181 		
182 	/**
183 		* <br /> Если в браузере консоль не поддерживается, 
184 		* <br /> используем альтернативный способ, собирая в буфер все данные
185 		* 
186 		* @param {boolean} [$debugToConsoleNotSupport=false]
187 		*  Использовать буфер для консоли и делать alert()
188 		* @return void
189 		*/
190 	initNotSupport:function($debugToConsoleNotSupport){
191 		if ($debugToConsoleNotSupport){
192 			/**
193 			 *  Создаём новый объект для работы с консолью
194 			 */
195 			window.console=new Object();
196 			var debug_buffer='';
197 			window.console.warn=function($str){
198 				alert('INFO:'+debug_buffer+"\n\r"+'WARN:'+"\n\r\t"+$str);
199 				debug_buffer='';
200 			};
201 			window.console.info=function($str){
202 				debug_buffer+="\n\r"+$str;
203 			};
204 			window.console.error=function($str){
205 				debug_buffer+="\n\rERROR:"+$str;
206 			};
207 		}
208 	}
209 	
210 };
211 }(document, jQuery);
