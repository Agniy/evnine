361 		function getParseURL($href,$tab_level){
362 			if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,$tab_level)+$EVNINE_NAME+'getParseURL($href='+$href+') BEGIN');
363 			if ($href===undefined){
364 				return false;
365 			}
366 			$href=$href.replace(/^.*\?/,"?");
367 			$reg = new RegExp($options.isHREFMatchThisRegExpSetSEFMode,"g");
368 			if ($href.match($reg)){
369 			/**
370 				*  Если совпадает с ЧПУ адресом
371 				*/
372 				$parse_url = this.getMethodFromSEFURL($href);
373 			}else {
374 			/**
375 				*  Используем плагин для разбора ссылки
376 				*/
377 				$parse_url = $.parseQuery($href);
378 			}
379 			if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,$tab_level+1)+$EVNINE_NAME+'$parse_url:');
380 			if ($options.debugToConsole) jQuery.evDev.getTraceObject($parse_url,jQuery.evDev.getTab($options.debugPrefixString,$tab_level+2));
381 			if (!$parse_url[$options.controller.paramName] || $parse_url[$options.controller.paramName]===null || $parse_url[$options.controller.paramName]==='' || $parse_url[$options.controller.paramName]==undefined){
382 				/**
383 					*  Проверяем указан ли контроллер в ссылке?
384 					*/
385 				$controller=$options.controller.defaultValue;
386 			}else {
387 			/**
388 				*  В случае если контроллер не указан,<br /> 
389 				*  используем значение по умолчанию из опций
390 				*/
391 				$controller=$parse_url[$options.controller.paramName];
392 			}
393 			if (!$parse_url[$options.method.paramName] || $parse_url[$options.method.paramName]===null || $parse_url[$options.method.paramName]==='' || $parse_url[$options.method.paramName]==undefined){
394 				/**
395 					*  Проверяем указан ли метод в ссылке?
396 					*/
397 				$method=$options.method.defaultValue;
398 			}else {
399 			/**
400 				*  В случае если метод не указан,<br /> 
401 				*  используем значение по умолчанию из опций
402 				*/
403 				$method=$parse_url[$options.method.paramName];
404 			}
405 			var $obj={};
406 			$obj[$options.controller.paramName]=$controller;
407 			$obj[$options.method.paramName]=$method;
408 			$obj.url=$href;
409 			if ($options.debugToConsole) jQuery.evDev.getTraceObject($obj,jQuery.evDev.getTab($options.debugPrefixString,$tab_level+1)+'return ');
410 			if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,$tab_level)+$EVNINE_NAME+'getParseURL($href='+$href+') END');
411 			return $obj;
412 		}
