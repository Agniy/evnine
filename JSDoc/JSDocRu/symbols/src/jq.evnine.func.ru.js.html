239 		/**
240 			*  Разберем адрес и получим метод и контроллер
241 			* 
242 			* @access private
243 			* @param $href - string
244 			* @param $tab_level - int  Отступы в режиме отладке
245 			* @return object{[$options.controller.paramName],$obj[$options.method.paramName]}
246 			*  
247 		*/
248 		function getParseURL($href,$tab_level){//Парсим адрес для получения шаблона и метода
249 			if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,$tab_level)+$EVNINE_NAME+'getParseURL($href='+$href+') BEGIN');
250 			if ($href===undefined){
251 				return false;
252 			}
253 			$href=$href.replace(/^.*\?/,"?");
254 			$reg = new RegExp($options.isHREFMatchThisRegExpSetSEFMode,"g");
255 			/* Если совпадает с ЧПУ адресом */
256 			if ($href.match($reg)){
257 				$parse_url = this.getMethodFromSEFURL($href);
258 			}else {
259 				$parse_url = $.parseQuery($href);
260 			}
261 			if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,$tab_level+1)+$EVNINE_NAME+'$parse_url:');
262 			if ($options.debugToConsole) jQuery.evDev.getTraceObject($parse_url,jQuery.evDev.getTab($options.debugPrefixString,$tab_level+2));
263 			/* Проверяем есть ли значение контроллера в адресе*/
264 			if (!$parse_url[$options.controller.paramName] || $parse_url[$options.controller.paramName]===null || $parse_url[$options.controller.paramName]==='' || $parse_url[$options.controller.paramName]==undefined){
265 				$controller=$options.controller.defaultValue;
266 			}else {
267 				/* В случае если имени контроллера нет, выставляем значение по умолчанию из опций*/
268 				$controller=$parse_url[$options.controller.paramName];
269 			}
270 			/* Проверяем есть ли значение метода в адресе */
271 			if (!$parse_url[$options.method.paramName] || $parse_url[$options.method.paramName]===null || $parse_url[$options.method.paramName]==='' || $parse_url[$options.method.paramName]==undefined){
272 				$method=$options.method.defaultValue;
273 			}else {
274 				/* В случае если имени метода нет, выставляем значение по умолчанию из опций*/
275 				$method=$parse_url[$options.method.paramName];
276 			}
277 			/* Возвращаем объект с значениями для последующего сохранения и получения функции */
278 			var $obj={};
279 			$obj[$options.controller.paramName]=$controller;
280 			$obj[$options.method.paramName]=$method;
281 			$obj.url=$href;
282 			if ($options.debugToConsole) jQuery.evDev.getTraceObject($obj,jQuery.evDev.getTab($options.debugPrefixString,$tab_level+1)+'return ');
283 			if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,$tab_level)+$EVNINE_NAME+'getParseURL($href='+$href+') END');
284 			return $obj;
285 		}

333 		/**
334 		 * Запустим функцию
335 		 * 
336 		 * @access private
337 		 * @param $function_callback - string
338 		 * @param $before - string
339 		 * @param $after - string
340 		 * @param $function_callback - string
341 		 * @return bool
342 		*/ 
343 		function setRunFunction($obj,$before,$after,$function_callback){
344 			if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,5)+$EVNINE_NAME+'setRunFunction() BEGIN');
345 			if ($obj==undefined){
346 				if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,5)+$EVNINE_NAME+'setRunFunction() END: return false, case:$obj==undefined');
347 				return false;
348 			}
349 			/* проверяем изменен ли параметр с прошлого раза */
350 			if ($before!==$after){
351 				if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,6)+'[$before='+$before+']!==[$after='+$after+']');
352 				/* проверяем уровень доступа */
353 				if (isHasAccess($obj)){
354 					$obj = $obj[$function_callback];
355 					if (typeof $obj==='function'){
356 						if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,5)+$EVNINE_NAME+'setRunFunction() END: return true, case:$obj===function');
357 						if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,6)+"$options.setJSFuncForLoadPage.setFunction."+$function_callback+"()");
358 						$obj();
359 						return true;
360 					}else {
361 						if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,5)+$EVNINE_NAME+'setRunFunction() END: return false, case:$before!==$after ');
362 						return false;
363 					}
364 				}
365 			}else {
366 				if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,6)+'[$before='+$before+']===[$after='+$after+']');
367 				if ($obj.setReloadPageAction==undefined){
368 					if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,5)+$EVNINE_NAME+'setRunFunction() END: return false, case:setReloadPageAction==undefined');
369 					return false;
370 				}
371 				/* проверяем уровень доступа */
372 				if (isHasAccess($obj)){
373 					if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,6)+"$options.setJSFuncForLoadPage.setFunction.setReloadPageAction()");
374 					$obj.setReloadPageAction();
375 					if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,5)+$EVNINE_NAME+'setRunFunction() END: return true');
376 					return true;
377 				}
378 			}
379 		}
