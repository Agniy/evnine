345 		/** getParseURL($href,$tab_level)<br />
346 			* 
347 			*  Obtain a method and controller of the URL
348 			* 
349 			* @param {string} $href
350 			*  Link to processing
351 			* 
352 			* @param {int} $tab_level
353 			*  Indentations in debug mode
354 			* 
355 			* @access private
356 			* @return {object} {$options.controller.paramName:function(),$options.method.paramName:function(),$url=string}
357 			* <br /> Returns an object for save and get the functions<br /> by controller and method name.
358 			*  
359 		*/
360 		function getParseURL($href,$tab_level){
361 			if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,$tab_level)+$EVNINE_NAME+'getParseURL($href='+$href+') BEGIN');
362 			if ($href===undefined){
363 				return false;
364 			}
365 			$href=$href.replace(/^.*\?/,"?");
366 			$reg = new RegExp($options.isHREFMatchThisRegExpSetSEFMode,"g");
367 			if ($href.match($reg)){
368 			/**
369 				*  If the SEF URL
370 				*/
371 				$parse_url = this.getMethodFromSEFURL($href);
372 			}else {
373 			/**
374 				*  Use the plugin to parse the references.
375 				*/
376 				$parse_url = $.parseQuery($href);
377 			}
378 			if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,$tab_level+1)+$EVNINE_NAME+'$parse_url:');
379 			if ($options.debugToConsole) jQuery.evDev.getTraceObject($parse_url,jQuery.evDev.getTab($options.debugPrefixString,$tab_level+2));
380 			if (!$parse_url[$options.controller.paramName] || $parse_url[$options.controller.paramName]===null || $parse_url[$options.controller.paramName]==='' || $parse_url[$options.controller.paramName]==undefined){
381 				/**
382 					*  Specify the controller in the link?
383 					*/
384 				$controller=$options.controller.defaultValue;
385 			}else {
386 			/**
387 				*  If the controller name is not specified,<br /> 
388 				*  use the default value of the options
389 				*/
390 				$controller=$parse_url[$options.controller.paramName];
391 			}
392 			if (!$parse_url[$options.method.paramName] || $parse_url[$options.method.paramName]===null || $parse_url[$options.method.paramName]==='' || $parse_url[$options.method.paramName]==undefined){
393 				/**
394 					*  Specify the method in the link?
395 					*/
396 				$method=$options.method.defaultValue;
397 			}else {
398 			/**
399 				*  If the method name is not specified,<br /> 
400 				*  use the default value of the options
401 				*/
402 				$method=$parse_url[$options.method.paramName];
403 			}
404 			var $obj={};
405 			$obj[$options.controller.paramName]=$controller;
406 			$obj[$options.method.paramName]=$method;
407 			$obj.url=$href;
408 			if ($options.debugToConsole) jQuery.evDev.getTraceObject($obj,jQuery.evDev.getTab($options.debugPrefixString,$tab_level+1)+'return ');
409 			if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,$tab_level)+$EVNINE_NAME+'getParseURL($href='+$href+') END');
410 			return $obj;
411 		}

488 			* @return {boolean}
489 			* <br /> true - if the function executed successfully.<br /> false - if function failed to call

491 		function setRunFunction($obj,$before,$after,$function_callback){
492 			if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,5)+$EVNINE_NAME+'setRunFunction() BEGIN');
493 			if ($obj==undefined){
494 			/**

498 				return false;
499 			}
500 			if ($before!==$after){
501 				/**
502 					*  Compare the method and controller before and after.
503 					*/
504 				if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,6)+'[$before='+$before+']!==[$after='+$after+']');
505 				if (isHasAccess($obj)){
506 					/**
507 						*  Check the level of access to the function of the options?
508 						*/
509 					$obj = $obj[$function_callback];
510 					if (typeof $obj==='function'){
511 						/**
512 							*  Executed if the object specified function.
513 							*/
514 						if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,5)+$EVNINE_NAME+'setRunFunction() END: return true, case:$obj===function');
515 						if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,6)+"$options.setJSFuncForLoadPage.setFunction."+$function_callback+"()");
516 						$obj();
517 						return true;
518 					}else {
