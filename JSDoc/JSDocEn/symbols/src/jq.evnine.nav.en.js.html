231 	function isHasCrossBrowserMinimumVersion() {
232 		if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,1)+$EVNINE_NAME+"isHasCrossBrowserMinimumVersion() BEGIN");
233 		var userAgent = navigator.userAgent.toLowerCase();
234 		jQuery.browser = {
235 			'version': (userAgent.match( /.+(?:rv|it|ra|ie|me)[\/: ]([\d.]+)/ ) || [])[1],
236 			'chrome': (/chrome/).test( userAgent ),
237 			'safari': (/webkit/).test( userAgent ) && !(/chrome/).test( userAgent ),
238 			'opera': (/opera/).test( userAgent ),
239 			'msie': (/msie/).test( userAgent ) && !(/opera/).test( userAgent ),
240 			'mozilla': (/mozilla/).test( userAgent ) && !(/(compatible|webkit)/).test( userAgent )
241 		};
242 		var $return=true;
243 		jQuery.each($options.crossBrowserMinimumVersion, function($browser,$version){
244 			if (jQuery.browser[$browser]){
245 				if (getInt(jQuery.browser.version)>=$version){
246 					if ($options.debugToConsole) console.warn("#getInt(jQuery.browser.version): "+getInt(jQuery.browser.version));
247 					if ($options.debugToConsole) console.warn("#$version: "+$version);
248 					if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,2)+$EVNINE_NAME+"isHasCrossBrowserMinimumVersion return true");
249 					$return=true;
250 					return '';
251 				}else {
252 					if ($options.debugToConsole) console.warn("#getInt(jQuery.browser.version): "+getInt(jQuery.browser.version));
253 					if ($options.debugToConsole) console.warn("#$version: "+$version);
254 					if ($options.debugToConsole) console.info(jQuery.evDev.getTab($options.debugPrefixString,2)+$EVNINE_NAME+"isHasCrossBrowserMinimumVersion return false");
255 					$return=false;
256 					return '';
257 				}
258 			}
259 		});
260 		if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,1)+$EVNINE_NAME+"isHasCrossBrowserMinimumVersion() END");
261 		return $return;
262 	}
263 	$options.isAllowThisBrowser=isHasCrossBrowserMinimumVersion();
264

401 		/**
402 			*  
403 			* 
404 			* @access public
405 			* @param $href - string
406 			* @param $method - string
407 			* @return $href string
408 			* 
409 		 */
410 		function setMethodTOURL ($href,$method) {
411 			if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,2)+$EVNINE_NAME+'setMethodTOURL() BEGIN');
412 			if ($method.length===0){
413 				return $href;
414 			}else {
415 				if ($href.match(/\.html$/)){
416 					$href = $href.substring(1);
417 					$method_replace  = $href.match(/.*\//);
418 					$method_replace=$method_replace.toString().split(/\//);
419 					if ($method_replace[1].match(/=/)){
420 						$method_match==$method_replace[0]+'/';
421 						$method='/'+$method_replace[0]+'/'+$method;
422 					}else {
423 						$method_match=$method_replace[1];
424 					}
425 					$href = $href.replace($method_match,$method);
426 				}else {
427 					$method_match = $href.match(/&m=.*/);
428 					$href = $href.replace($method_match,"&m="+$method);
429 				}
430 				if ($options.debugToConsole) console.warn(jQuery.evDev.getTab($options.debugPrefixString,2)+$EVNINE_NAME+'setMethodTOURL() END');
431 				return $href;
432 			}
433 		}
